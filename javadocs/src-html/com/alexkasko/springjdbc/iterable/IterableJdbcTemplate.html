035        /**
036         * Static method to use in finally methods for closing
037         * {@link CloseableIterator}s. Writes warning into log on exception.
038         *
039         * @param iter iterator to close
040         */
041        public static void closeQuetly(CloseableIterator<?> iter) {
042            try {
043                if (iter != null) {
044                    iter.close();
045                }
046            } catch (Exception e) {
047                logger.warn("Error on closing iterator: [" + iter + "]", e);
048            }
049        }

059        /**
060         * {@inheritDoc}
061         */
062        @Override
063        // see {@code JdbcTemplate#query(org.springframework.jdbc.core.PreparedStatementCreator, org.springframework.jdbc.core.PreparedStatementSetter, org.springframework.jdbc.core.ResultSetExtractor)}
064        // see {@code JdbcTemplate#execute(org.springframework.jdbc.core.PreparedStatementCreator, org.springframework.jdbc.core.PreparedStatementCallback)}
065        // see {@code JdbcTemplate#query(org.springframework.jdbc.core.PreparedStatementCreator, org.springframework.jdbc.core.PreparedStatementSetter, org.springframework.jdbc.core.ResultSetExtractor)}
066        public <T> CloseableIterator<T> queryForIter(PreparedStatementCreator psc, PreparedStatementSetter pss,
067                                                     RowMapper<T> rowMapper) throws DataAccessException {
068            Assert.notNull(psc, "PreparedStatementCreator must not be null");
069            Assert.notNull(rowMapper, "RowMapper must not be null");
070            if(logger.isDebugEnabled()) {
071                String sql = getSql(psc);
072                logger.debug("Executing prepared SQL statement" + (sql != null ? " [" + sql + "]" : ""));
073            }
074            DataSource ds = getDataSource();
075            Connection con = DataSourceUtils.getConnection(ds);
076            PreparedStatement ps = null;
077            ResultSet rs = null;
078            try {
079                Connection conToUse = con;
080                NativeJdbcExtractor nje = getNativeJdbcExtractor();
081                if(nje != null && nje.isNativeConnectionNecessaryForNativePreparedStatements()) {
082                    conToUse = nje.getNativeConnection(con);
083                }
084                ps = psc.createPreparedStatement(conToUse);
085                applyStatementSettings(ps);
086                PreparedStatement psToUse = ps;
087                if(nje != null) {
088                    psToUse = nje.getNativePreparedStatement(ps);
089                }
090                if(pss != null) {
091                    pss.setValues(psToUse);
092                }
093                rs = psToUse.executeQuery();
094                ResultSet rsToUse = rs;
095                if(nje != null) {
096                    rsToUse = nje.getNativeResultSet(rs);
097                }
098                // warnings are handled after query execution but before data access
099                handleWarnings(ps);
100                return new PreparedStatementCloseableIterator<T>(ds, con, psc, pss, ps, rs, rsToUse, rowMapper);
101            } catch(SQLException ex) {
102                // Release Connection early, to avoid potential connection pool deadlock
103                // in the case when the exception translator hasn't been initialized yet.
104                if(psc instanceof ParameterDisposer) {
105                    ((ParameterDisposer) psc).cleanupParameters();
106                }
107                String sql = getSql(psc);
108                psc = null;
109                JdbcUtils.closeResultSet(rs);
110                rs = null;
111                JdbcUtils.closeStatement(ps);
112                ps = null;
113                DataSourceUtils.releaseConnection(con, getDataSource());
114                con = null;
115                throw getExceptionTranslator().translate("PreparedStatementCallback", sql, ex);
116            }
117    //        resources will be closed in iterator
118    //        finally {
119    //            if(psc instanceof ParameterDisposer) {
120    //                ((ParameterDisposer) psc).cleanupParameters();
121    //            }
122    //            JdbcUtils.closeStatement(ps);
123    //            DataSourceUtils.releaseConnection(con, getDataSource());
124    //        }
125        }

127        /**
128         * {@inheritDoc}
129         */
130        @Override
131        public <T> CloseableIterator<T> queryForIter(String sql, RowMapper<T> rowMapper) throws DataAccessException {
132            Assert.hasText(sql, "Provided sql query is blank");
133            Assert.notNull(rowMapper, "RowMapper must not be null");
134            DataSource ds = getDataSource();
135            Connection con = DataSourceUtils.getConnection(ds);
136            Statement stmt = null;
137            ResultSet rs = null;
138            try {
139                Connection conToUse = con;
140                NativeJdbcExtractor nje = getNativeJdbcExtractor();
141                if(nje != null && nje.isNativeConnectionNecessaryForNativeStatements()) {
142                    conToUse = nje.getNativeConnection(con);
143                }
144                stmt = conToUse.createStatement();
145                applyStatementSettings(stmt);
146                Statement stmtToUse = stmt;
147                if(nje != null) {
148                    stmtToUse = nje.getNativeStatement(stmt);
149                }
150                rs = stmtToUse.executeQuery(sql);
151                ResultSet rsToUse = rs;
152                if(nje != null) {
153                    rsToUse = nje.getNativeResultSet(rs);
154                }
155                // warnings are handled after query execution but before data access
156                handleWarnings(stmt);
157                return new StatementCloseableIterator<T>(ds, con, stmt, rs, rsToUse, rowMapper);
158            } catch(SQLException ex) {
159                JdbcUtils.closeResultSet(rs);
160                rs = null;
161                // Release Connection early, to avoid potential connection pool deadlock
162                // in the case when the exception translator hasn't been initialized yet.
163                JdbcUtils.closeStatement(stmt);
164                stmt = null;
165                DataSourceUtils.releaseConnection(con, getDataSource());
166                con = null;
167                throw getExceptionTranslator().translate("StatementCallback", getSql(sql), ex);
168            }
169    //        resources will be closed in iterator
170    //              finally {
171    //                      JdbcUtils.closeStatement(stmt);
172    //                      DataSourceUtils.releaseConnection(con, getDataSource());
173    //              }
174        }

264        /**
265         * Determine SQL from potential provider object.
266         * Borrowed from {@code JdbcTemplate}
267         *
268         * @param sqlProvider object that's potentially a SqlProvider
269         * @return the SQL string, or <code>null</code>
270         * @see SqlProvider
271         */
272        private static String getSql(Object sqlProvider) {
273            if(sqlProvider instanceof SqlProvider) {
274                return ((SqlProvider) sqlProvider).getSql();
275            } else {
276                return null;
277            }
278        }

280        /**
281         * Simple adapter for PreparedStatementCreator, allowing to use a plain SQL statement.
282         * Borrowed from {@code JdbcTemplate}
283         */
284        private static class SimplePreparedStatementCreator implements PreparedStatementCreator, SqlProvider {
285            private final String sql;
286            public SimplePreparedStatementCreator(String sql) {
287                Assert.notNull(sql, "SQL must not be null");
288                this.sql = sql;
289            }
290            public PreparedStatement createPreparedStatement(Connection con) throws SQLException {
291                return con.prepareStatement(this.sql);
292            }
293            public String getSql() {
294                return this.sql;
295            }
296        }
